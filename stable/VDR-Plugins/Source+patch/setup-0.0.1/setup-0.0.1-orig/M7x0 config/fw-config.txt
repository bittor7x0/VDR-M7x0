#----- Video und LED


#PARAMETER: CONFIG_VIDEO_TV_MODE key undef svideo fbas
#
# Video-Modus fuer den TV-Ausgang (SCART). Die Voreinstellung 'undef' ist dabei
# ein Hinweis fuer die Wavebox-Migration, dass der Modus noch nicht gesetzt wurde.

#

#PARAMETER: CONFIG_LED_BRIGHTNESS num 40 0 100
#
# Dieser Wert gibt die LED-Helligkeit der aeusseren beiden LEDs an, wenn diese von
# der Lemmi Firmware angesteuert werden.
#
# Der Wert von 0 bedeutet, dass moeglichst alle LEDs, auch die mittler LED, aus
# bleiben. Werte zwischen 1 und 40 legen die Helligkeit fest. Wert >40 machen
# erhoehen die Helligkeit nicht mehr. Sie haben aber einen Effekt auf das
# Zeitverhalten beim Dimmen.
#
# Zum Testen kann bei angehaltenem Boot-Menu das Kommando 'picctl dimm-led 0 x 2
# 0x11' verwendet werden. Fuer 'x' sind dann Werte zwischen 1 und 100 einzutragen.
#
# Die Voreinstellung ist '40'.
#
CONFIG_LED_BRIGHTNESS = 40

# Die Voreinstellung ist 'no'.
#
CONFIG_LOG_KMSG = no


#----- USB


#PARAMETER: CONFIG_USB_AUTO_MOUNT str
#
# Dieser Parameter legt die USB-Laufwerke fest, die automatisch mit dem
# Systemstart geladen werden. Es wird eine mir Lerrzeichen getrennte
# Zeichenkette erwartet, wobei Gross-/Kleinschreibung unerheblich ist.
#
# Die USB-Platten (oder Sticks) werden den Ports zugeordnet und heissen 'usb1'
# und 'usb2'. Fuer jede dieser Platten werden bis zu 26 Partitonen vom Typ
# 'ext2/3' und 'FAT' unterstuetzt. Die erste gueltige Partition erhaelt den
# Kennbuchstaben 'a', die naechste 'b' und so fort.
#
# Die Angabe 'usb1 usb2c' bedeutet nun, das alle Partition auf der USB-Platte
# an Port #1 und die dritte Partition auf der USB-Platte an Port #2 automatisch
# geladen werden soll. Die Prefixe 'usb' duerfen uebrigens weggelassen werden,
# so dass 'usb1 usb2c' kuerzer als '1 2c' geschrieben werden darf.
#
# Eine leere Zeichenkette (die Voreinstellung) oder das Kennwort 'ALL'
# bedeuten, dass alle erkannten USB-Partitionen gemounted werden.
#
CONFIG_USB_AUTO_MOUNT = 


#----- Netzwerk


#PARAMETER: CONFIG_NET_MODE key undef dhcp ip off
#
# Hiermit wird der Netzwerkmodus einegstellt:
# * undef: Ein Hinweis fuer die Wavebox-Migration, dass der
# Modus noch nicht gesetzt wurde. Es wird aber DHCP verwendet.
# * dhcp: DHCP wird verwendet.
# * ip: Die IP-Werte der Parameter CONFIG_IP und CONFIG_NETMASK
# sollen verwendet werden. Wird erkannt, dass einer der beiden
# Werte ungueltig ist, dann wird ersatzweise DHCP verwendet.
# * off: Kein Netzwerk initialisieren.
#
# Die Voreinstellung ist 'undef'.
#
CONFIG_NET_MODE = ip

#PARAMETER: CONFIG_NETMASK str

#

# Voreinstellung verwendet. Weitere Programm moegen dieses auch tun.
#
# Dieser Parameter legt den SMTP-Server fest, an dem die Mails weitergeleitet

# die Werte CONFIG_MAIL_SERVER, CONFIG_MAIL_FROM und CONFIG_MAIL_REPLY als
# Voreinstellung verwendet. Weitere Programm moegen dieses auch tun.
#
# Dieser Parameter legt den Absender fest. Ohne Angabe verwendet das Script
# 'mail' den Namen des angemeldeten Nutzers und den Rechnernamen, den das

#
CONFIG_MAIL_FROM = 

#PARAMETER: CONFIG_MAIL_REPLY str
#
# Das Mail-Programm 'mini_sendmail' kann Mails an einen MTA (Mail Transfer Agent)
# weitergeben. Als User-Frontend gibt es das Script 'mail', welches automatisch
# die Werte CONFIG_MAIL_SERVER, CONFIG_MAIL_FROM und CONFIG_MAIL_REPLY als
# Voreinstellung verwendet. Weitere Programm moegen dieses auch tun.
#
# Dieser Parameter legt die Antwortadresse ("Reply-To:") fest. Dieses koennte
# wichtig sein, weil die Box keine Mails empfangen kann. Ohne Angabe wird keine
# Antwortadresse angegeben.
#
CONFIG_MAIL_REPLY = 


#----- Menüs


#PARAMETER: CONFIG_STARTMENU_MODE key normal fast off
#
# Mit dieser Option wird festgelegt, ob das Start-Menu beim Systemstart
# angezeigt werden soll. In der Einstellung 'off' wird es nicht angezeigt.
#
# In den anderen beiden Einstellungen wird es angezeigt, wobei nach einer
# gewissen Zeit der VDR gemaess Parameter CONFIG_VDR_STANDARD gestartet wird.
# In der Einstellung 'fast' wird CONFIG_STARTMENU_SHORT_WAIT Sekunden und in
# der Einstellung 'normal' wird CONFIG_STARTMENU_WAIT Sekunden gewartet.
#
# Unabhaengig von dieser Einstellung wird das Start-Menu bei einem
# Firmware-Wechsel oder bei Problemen angezeigt.
#
# Die Voreinstellung ist 'normal'.
#
CONFIG_STARTMENU_MODE = normal

#PARAMETER: CONFIG_STARTMENU_WAIT num 30 5 300
#
# Diese Option zeigt nur Wirkung bei aktiviertem Boot-Menu. Sie
# gibt die Wartezeit in Sekunden an, bis die Standardkonfiguration
# (siehe CONFIG_START_MODE) automatisch gestartet wird. Dieses ist
# wichtig fuer den Timer-gesteuerten Start der Box.
#
# Die Voreinstellung ist '30'.
#
CONFIG_STARTMENU_WAIT = 30

#PARAMETER: CONFIG_STARTMENU_SHORT_WAIT num 5 5 300
#
# Es gibt Situationen, da soll das Boot-Menu nur kurz
# aktiviert werden, bevor der Auto-Start durchgefuehrt wird.
# Dieswer Paramweter gibt die Wartezeit in Sekunden an.
#
# Die Voreinstellung ist '5'.
#
CONFIG_STARTMENU_SHORT_WAIT = 5

#PARAMETER: CONFIG_HALTMENU_ENABLED key yes no
#
# Mit dieser Option wird das Shutdown-Menu aktiviert. Im Shutdown-Menu
# kann der Benutzer nach der Verwendung des VDRs oder der wavebox
# nochmals eingreifen, bevor das System entgueltig abgeschaltet wird.
#
# Beachtet bitte auch den Parameter CONFIG_SET_VIDEO_MODE. In dessen
# Voreinstellung 'interactive' wird das Shutdown-Menu nur angezeigt,
# wenn eine interaktive Eingabe erforderlich ist.
#
# Die Voreinstellung ist 'yes'.
#
CONFIG_HALTMENU_ENABLED = yes

#PARAMETER: CONFIG_HALTMENU_WAIT num 30 5 300
#
# Diese Option zeigt nur Wirkung bei aktiviertem Shutdown-Menu. Sie gibt die
# Wartezeit in Sekunden an, bis der automatische Shutdown ausgeloest wird.
#
# Die Voreinstellung ist '30'.
#
CONFIG_HALTMENU_WAIT = 5

#PARAMETER: CONFIG_MENU_HIDE num 90 0 600
#
# Falls >0, dann werden die Menus durch einen schwarzen Bildschirm
# ersetzt, wenn die angegebene Anzahl von Sekunden keine Benutzereingabe
# erfolgte. Die Zeitbeginnt erst nach dem Ende eines evtl. Count-Downs.
#
# Die Voreinstellung ist '90'.
#
CONFIG_MENU_HIDE = 90


#----- VDR


#PARAMETER: CONFIG_VDR_STANDARD key vdr1 vdr2 vdr3 vdr4
#
# Neben dem internen VDR (VDR #1) werden 3 externe VDR (#2 bis #4)
# unterstuetzt. Mit diesem Parameter wird entschieden, welches dieser vier
# Varianten der Standard-VDR ist und u.a. bei Timer-Aufnahmen aufgerufen wird.
#
# Die Voreinstellung ist 'vdr1', der interne VDR.
#
CONFIG_VDR_STANDARD = vdr4

#PARAMETER: CONFIG_VDR1_LOG key none net
#
# Wohin soll das Logging des internen VDRs (VDR1) geschrieben werden:
#
# * none: Es wird kein Logging durchgefuehrt.
# * net: Die Log-Ausgabe wird via stdin2tcp an TCP Port 12007 ausgegeben.
#
# Die Voreinstellung ist 'file'. Diese Einstellung gilt fuer den internen VDR.
#
CONFIG_VDR1_LOG = none

#PARAMETER: CONFIG_VDR_VIDEO_PATH str
#
# Dieser Parameter legt den Pfad zum VDR-Aufnahme-Verzeichnis fuer die drei
# externen Konfigurationen VDR #2, #3 und #4 fest, z.B: "/media/usb1a/vdr/videos"
# oder "videos". Ohne Pfadangabe wird der jeweilige VDR_PATH verwendet.
#
# Beginnt die Pfadangabe nicht mit einem Schraegstrich ('/'), dann wird der Pfad
# relativ zu CONFIG_VDR_PATH interpretiert.
#
# Diese Einstellung gilt gemeinsam fuer die drei externen VDR #2 bis #4. Die
# Voreinstellung ist die leere Zeichenkette.
#
CONFIG_VDR_VIDEO_PATH = /media/pc2/vdrVideo

#PARAMETER: CONFIG_VDR_EPG_PATH str
#
# Dieser Parameter legt den Pfad zum VDR-EPG-Verzeichnis fuer die drei externen
# Konfigurationen VDR #2, #3 und #4 fest, z.B: "/media/usb1a/vdr/epg" oder "epg".
# Ohne Pfadangabe wird das errechnete Konfigurationsverzeichnis verwendet.
#
# Beginnt die Pfadangabe nicht mit einem Schraegstrich ('/'), dann wird der Pfad
# relativ zu CONFIG_VDR_PATH interpretiert.
#
# Diese Einstellung gilt gemeinsam fuer die drei externen VDR #2 bis #4. Die
# Voreinstellung ist die leere Zeichenkette.
#
CONFIG_VDR_EPG_PATH = 

#PARAMETER: CONFIG_VDR_N_LOG_BACKUP num 3 0 20
#
# Der VDR erzeugt bei jedem Aufruf eine Log-Datei, wenn durch den
# Parameter CONFIG_VDR_LOG die Ausgabe auf Datei aktiviert ist.
#
# Fuer spaetere Analysen ist es evtl. wichtig, dass eine solche
# Log-Datei nicht sofort wieder ueberschrieben wird. Daher koennen
# zirkulaere Sicherheitskopien der Log-Datei erstellt werden. Mit
# diesem Parameter wird die Anzahl der Sicherheitskopien eingestellt.
#
# Diese Einstellung gilt gemeinsam fuer die drei externen VDR #2 bis
# #4. Die Voreinstellung ist '3'.
#
CONFIG_VDR_N_LOG_BACKUP = 20


#----- VDR #2


#PARAMETER: CONFIG_VDR2_NAME str
#
# Neben dem internen VDR (VDR #1) werden 3 externe VDR (#2 bis #4)
# unterstuetzt. Um diese externe VDR auseinander zuhalten, kann
# ihnen ein Name zugewiesen werden. Dieser Parameter legt den Namen

#
# Die Voreinstellung ist die leere Zeichenkette.

CONFIG_VDR2_NAME = intern, mit Video

#PARAMETER: CONFIG_VDR2_PATH str
#
# Dieser Parameter legt den Pfad zum VDR-Verzeichnis oder zum VDR-Programm fuer
# den VDR #2 (VDR #1 ist der interne VDR, ab #2 sind es die exteren VDRs) fest.
# Enthaelt die Parameterangabe den Namen eines Verzeichnisses, dann wird
# automatisch das Dateiname 'vdr' ergaenzt.
#
# Dieser Parameter wird auch fuer die VDR-Installation durch ein o7o-Archiv
# verwendet und sollte daher definitiv auf das VDR-Programm zeigen. Moechte man
# seinen VDR anders, z.B. ueber ein eigenes Script, starten, dann kann dieses
# Script mit dem Parameter CONFIG_VDR_STARTER angegeben werden.
#
# Der besondere Wert 'Stern' ('*') bedeutet, dass der interne VDR verwendet
# werden soll, wobei Starter und Konfigurationsverzeichnis aus den den beiden
# Parametern CONFIG_VDR2_STARTER und CONFIG_VDR2_CONFIG_PATH ermittelt wird.
#
# Die Voreinstellung ist die leere Zeichenkette, die bedeutet, dass kein VDR
# installiert ist. Beispiel: "/media/usb1a/vdr" oder "/media/pc2/vdr/vdr-prog".
#
CONFIG_VDR2_PATH = *

#PARAMETER: CONFIG_VDR2_STARTER str
#
# Ist dieser Parameter nicht leer und existiert das angegebene Script, dann
# wird der VDR nicht direkt aufgerufen, sondern ersatzweise dieses
# Start-Script. Damit hat man die Aufruf-Kontrolle des VDR und kann u.a. die
# Plugins konfigurieren. Diese Einstellung gilt fuer den VDR #2.
#
# Pfadangaben, die nicht mit einem '/' beginnen, werden relativ zum
# Verzeichnis von CONFIG_VDR2_PATH interpretiert.
#
# Bei jedem VDR-Start wird die Datei 'vdr-starter.template' angelegt bzw.
# ueberschrieben, die als Vorlage fuer ein eigenes Start-Script dienen kann.
#
# Die Voreinstellung ist die leere Zeichenkette.
#
CONFIG_VDR2_STARTER = 

#PARAMETER: CONFIG_VDR2_CONFIG_PATH str
#
# Dieser Parameter legt den Pfad zum VDR-Aufnahme-Verzeichnis
# fuer den VDR #2 fest.
#
# Pfadangaben, die nicht mit einem '/' beginnen, werden relativ
# zum Verzeichnis von CONFIG_VDR2_PATH interpretiert.
#
# In der Voreinstellung ohne Pfadangabe wird ein Unterverzeichnis
# von CONFIG_VDR2_PATH verwendet, welches den Hostnamen und die
# MAC beruecksichtigt, damit mehrere Boxen mit gemeinsamen
# VDR-Verzeichnis eigene Konfigurationsverzeichnisse erhalten.
#
# Der besondere Wert 'Stern' ('*') bedeutet, dass die
# Kofiguration des internen VDRs verwendet werden soll.
#
# Beispiel: "/media/usb1a/vdr-config" oder "conf-wohnzimmer".
#
CONFIG_VDR2_CONFIG_PATH = *

#PARAMETER: CONFIG_VDR2_LOG key file net both none
#
# Wohin soll das Logging des VDR geschrieben werden:
#
# * file: Die Log-Ausgabe wird in die Datei vdr.log geschrieben.
# * net: Die Log-Ausgabe wird via stdin2tcp an TCP Port 12007 ausgegeben.
# * both: Sowohl 'file' als auch 'net' werden aktiviert.
# * none: Es wird kein Logging durchgefuehrt. Diese Einstellung gilt fuer
# den VDR #2.
#
# Die Voreinstellung ist 'file'.
#
CONFIG_VDR2_LOG = file

#PARAMETER: CONFIG_VDR4_STARTER str
#
# Ist dieser Parameter nicht leer und existiert das angegebene Script, dann
# wird der VDR nicht direkt aufgerufen, sondern ersatzweise dieses
# Start-Script. Damit hat man die Aufruf-Kontrolle des VDR und kann u.a. die
# Plugins konfigurieren. Diese Einstellung gilt fuer den VDR #4.
#
# Pfadangaben, die nicht mit einem '/' beginnen, werden relativ zum
# Verzeichnis von CONFIG_VDR4_PATH interpretiert.
#
# Bei jedem VDR-Start wird die Datei 'vdr-starter.template' angelegt bzw.
# ueberschrieben, die als Vorlage fuer ein eigenes Start-Script dienen kann.
#
# Die Voreinstellung ist die leere Zeichenkette.
#
CONFIG_VDR4_STARTER = 

#PARAMETER: CONFIG_VDR4_CONFIG_PATH str
#
# Dieser Parameter legt den Pfad zum VDR-Aufnahme-Verzeichnis
# fuer den VDR #4 fest.
#
# Pfadangaben, die nicht mit einem '/' beginnen, werden relativ
# zum Verzeichnis von CONFIG_VDR4_PATH interpretiert.
#
# In der Voreinstellung ohne Pfadangabe wird ein Unterverzeichnis
# von CONFIG_VDR4_PATH verwendet, welches den Hostnamen und die
# MAC beruecksichtigt, damit mehrere Boxen mit gemeinsamen
# VDR-Verzeichnis eigene Konfigurationsverzeichnisse erhalten.
#
# Der besondere Wert 'Stern' ('*') bedeutet, dass die
# Kofiguration des internen VDRs verwendet werden soll.
#
# Beispiel: "/media/usb1a/vdr-config" oder "conf-wohnzimmer".
#
CONFIG_VDR4_CONFIG_PATH = 

#PARAMETER: CONFIG_VDR4_LOG key file net both none
#
# Wohin soll das Logging des VDR geschrieben werden:
#
# * file: Die Log-Ausgabe wird in die Datei vdr.log geschrieben.
# * net: Die Log-Ausgabe wird via stdin2tcp an TCP Port 12007 ausgegeben.
# * both: Sowohl 'file' als auch 'net' werden aktiviert.
# * none: Es wird kein Logging durchgefuehrt. Diese Einstellung gilt fuer
# den VDR #4.
#
# Die Voreinstellung ist 'file'.
#
CONFIG_VDR4_LOG = file


#----- BoxAmp


#PARAMETER: CONFIG_BOXAMP_PATH str
#
# Dieser Parameter legt den Pfad zum BoxAmp-Verzeichnis oder zum BoxAmp-Programm
# fest. Ohne Angabe wird davon ausgegangen, dass kein BoxAmp installiert ist.
#
# Enthaelt die Parameterangabe den Namen eines Verzeichnisses, dann wird
# automatisch das Dateiname 'boxamp' ergaenzt.
#
# Die Voreinstellung ist die leere Zeichenkette. Beispiele: "/media/usb1a/mp3"
# oder "/media/pc2/mp3/boxamp".
#
CONFIG_BOXAMP_PATH = /media/USB-HDD-1/mp3Player/boxamp

#PARAMETER: CONFIG_BOXAMP_MP3_PATH str
#
# Dieser Parameter legt das mp3-Basisverzeichnis fuer den
# BoxAmp fest. Der BoxAMp spielt nur mp3-Dateien aus dem
# angegeben Verzeichnis und aus dessen Unterverzeichnissen ab.
#
# Die Voreinstellung ist die leere Zeichenkette.
#
CONFIG_BOXAMP_MP3_PATH = /pc1

#PARAMETER: CONFIG_BOXAMP_OPTIONS str
#
# Mit Diesem Parameter koennen dem Programm
# BoxAmp zusaetzliche Optiknen uebergeben werden.
#
# Die Voreinstellung ist die leere Zeichenkette.
#
CONFIG_BOXAMP_OPTIONS = -d

#PARAMETER: CONFIG_LAST_WSW_PATH str
#
# Bei der Installation eines Updates ([WSW-Datei]]) wird hier der komplette
# Pfad der Datei eingetragen, damit dort die naechste WSW-Suche beginnen kann.
#
# Die Voreinstellung ist eine leere Zeichenkette.
#
CONFIG_LAST_WSW_PATH = 

#PARAMETER: CONFIG_LAST_O7O_PATH str
#
# Bei der Ausfuehrung eines [o7o-Archiv|o7o-Archives]] wird hier der kpomplette
# Pfad der Datei eingetragen, damit dort die naechste o7o-Suche beginnen kann.
#
# Die Voreinstellung ist eine leere Zeichenkette.
#
CONFIG_LAST_O7O_PATH = 

#PARAMETER: CONFIG_WB_MIGRATION_DONE key no yes
#
# Die Firmware kann beim Systemstart versuchen, fehlende Einstellungen aus den
# Wavebox-Einstellungen zu migrieren. Dieses ist vor allen beim ersten Start einer
# neuen Konfiguration wichtig.
#
# In der Voreinstellung 'no' wird diese Migration ausgefuehrt. Es reicht aber die
# einmalige Migration, so dass dieser Parameter dann automatisch auf 'yes' gesetzt
# wird, um ab dem naechsten Systemstart die dann unnoetige Analyse auszulassen.
#
CONFIG_WB_MIGRATION_DONE = yes
